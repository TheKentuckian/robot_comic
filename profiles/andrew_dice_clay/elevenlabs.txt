# ElevenLabs TTS configuration for the Andrew Dice Clay profile.
# Format: key=value (one per line; comments start with #).

# Voice name: must match a voice in your ElevenLabs catalog. Dice lands best
# on a voice with Brooklyn-bred swagger and mid-low grit — conviction in every
# syllable, never soft. Closest prebuilt options:
#   Brian  — deep, resonant, comforting   (shares register with Hicks; fits the
#             baritone rasp Dice delivers on the verdict-style punchlines)
#   Adam   — dominant, firm, slightly deeper  (more authority, less warmth)
#
# Chosen: Brian — the warm-but-resonant baseline lets the swagger land without
# sounding purely aggressive, which would flatten Dice's mock-outrage character.
# Swap to Adam if you want more "pronouncement" energy over "entertainer" energy.
voice=Brian

# No IVC/PVC clone committed here. If you train an Andrew Dice Clay clone,
# add a sibling elevenlabs.local.txt (gitignored) with:
#   voice_id=<your_ivc_or_pvc_voice_id>
# That file overrides this one at runtime.

# Voice stability: LOW — Dice's delivery rockets between the slow strut and the
# sudden barked punchline. Too much stability kills the swagger-to-snap contrast.
stability=0.30

# Similarity boost: high to keep Brian's character intact through the wide
# stability swings, so identity stays consistent even at the loud peaks.
similarity_boost=0.75
